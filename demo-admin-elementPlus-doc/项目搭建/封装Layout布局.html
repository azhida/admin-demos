Layout 组件组成部分 ​

  • 头部组件 Header ： <el-header> 顶栏容器
  • 侧边栏组件 Aside : <el-aside> 侧边栏容器
  • 内容区域 Main : <el-main> 主要区域容器
  • 底栏 Footer : <el-footer> 底栏容器
  • 页签组件 TagsView

  • src/layout/components/Aside/SubMenu.vue
示例代码
vue
<template>
    <template v-for="e in menuList" :key="e.path">
        <el-sub-menu :index="e.path"  v-if="e.children && e.children.length > 0" >
            <template #title>
                <el-icon v-if="e.icon"><component :is="e.icon"></component></el-icon>
                {{e.title}}
            </template>
            <SubMenu :menu-list="e.children"></SubMenu>
        </el-sub-menu>
        <el-menu-item :index="e.path" v-else>            
            <el-icon v-if="e.icon"><component :is="e.icon"></component></el-icon>
            {{e.title}}
        </el-menu-item>
    </template>
</template>

<script setup lang="ts">

interface Menu {
    id?:number|string
    path:string
    title: string
    icon: string
    children?:Array<Menu>
}

interface Prop {
    menuList: Menu[]
}

defineProps<Prop>()
</script>
<template>
    <template v-for="e in menuList" :key="e.path">
        <el-sub-menu :index="e.path"  v-if="e.children && e.children.length > 0" >

</template>
